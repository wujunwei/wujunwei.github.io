func majorityElement(nums []int) int {
	length := len(nums)
	count := make(map[int]int)
	for _, value := range nums {
		count[value]++
		if count[value] > length/2 {
            return value
		}
	}
	return -1
}

package main
func majorityElement(nums []int) int {
	count := 1
	index := nums[0]
	for i := 1 ; i < len(nums) ; i++{
		if nums[i] == index{
			count++
		}else{
			count--
			if count == 0{
				index = nums[i + 1]
			}
		}
	}
	return index
}
