package main
type matrix [2][2]int

func multiply(a, b matrix) (c matrix) {
	for i := 0; i < 2; i++ {
		for j := 0; j < 2; j++ {
			c[i][j] = (a[i][0]*b[0][j] + a[i][1]*b[1][j])%(1e9+7)
		}
	}
	return
}

func pow(a matrix, n int) matrix {
	ret := matrix{{1, 0}, {0, 1}}
	for ; n > 0; n >>= 1 {
		if n&1 == 1 {
			ret = multiply(ret, a)
		}
		a = multiply(a, a)
	}
	return ret
}

package main
func pow(a float64, n int) float64 {
	ret := 1
	for ; n > 0; n >>= 1 {
		if n&1 == 1 {
			ret = a
		}
		a = a * a
		for a > 1e9+7 {
			a -= 1e9 + 7
		}
	}
	return ret
}

func fib(n int) int {
    sqrt5 := math.Sqrt(5)
    p1 := pow((1+sqrt5)/2, float64(n))
    p2 := pow((1-sqrt5)/2, float64(n))
    return int(math.Round((p1 - p2) / sqrt5))
}
